/* 3)   Push all the zeroÂ’s of a given array to the end of the array */
 #include<stdio.h>
 int pushzeroestoend(int arr[100],int n)
 {
	int count=0,no_of_zeroes;
  	for(int i=0;i<n;i++)
  	{
    		if(arr[i]!=0)
    		{
      			arr[count]=arr[i];
      			count=count+1;
    		}
  	}
  	no_of_zeroes=n-count;
  	while(no_of_zeroes)
  	{
    		arr[count]=0;
    		count++;
    		no_of_zeroes--;
  	}
  	printf("\nthe altered array is ");
  	for(int k=0;k<n;k++)
    		printf("%d\t",arr[k]);
}
